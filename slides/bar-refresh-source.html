var timerId = null;
$(mySection).on('start', function (event, args) {
    if (timerId === null) {
        timerId = setInterval(function () {
            data = refreshData()
            chart.update(data)
        },args.rate)
    }
})

$(mySection).on('stop', function () {
    if (timerId !== null) {
        clearInterval(timerId);
        timerId = null;
    }
})
